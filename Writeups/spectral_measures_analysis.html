spectral_measures <- 
  read_csv("02_PossibleRecordings/SpectralMeasuresTextgrids/spectral_measures.txt") |> 
  rename(file = Filename,
         utterance = var1,
         session = var2,
         tune = var3,

  mutate(wordset = ifelse(utterance %in% two_syl_words, "syl2", "syl3"),

  filter(f0 > 0, f0 < 200) |>
